                                        <p><strong>Marca:</strong> ${selectedVehicle.marca}</p>
                                        <p><strong>Ano:</strong> ${selectedVehicle.ano}</p>
                                        <p><strong>Quilometragem:</strong> ${selectedVehicle.km.toLocaleString()} km</p>
                                        <p><strong>Combustível:</strong> ${selectedVehicle.combustivel}</p>
                                        <p><strong>Câmbio:</strong> ${selectedVehicle.cambio}</p>
                                        <p><strong>Cor:</strong> ${selectedVehicle.cor}</p>
                                    </div>
                                </div>
                                <div>
                                    <h2 class="text-xl font-semibold mb-4" style="color:${isDarkMode ? '#e2e8f0' : '#1f2937'}">Descrição</h2>
                                    <p>${selectedVehicle.descricao}</p>
                                </div>
                            </div>
                            <button onclick="handleContact(${JSON.stringify(selectedVehicle).replace(/"/g, '&quot;')})" class="w-full mt-8 bg-green-500 text-white py-4 rounded-lg font-semibold text-lg hover:bg-green-600 transition">
                                Tenho Interesse
                            </button>
                        </div>
                    </div>
                </div>
            `;
        }
        else if (currentPage === 'admin' && isAdmin) {
            html += `
                <div class="container mx-auto px-4 py-8">
                    <div class="flex justify-between items-center mb-8">
                        <h1 class="text-3xl font-bold" style="color:${isDarkMode ? '#e2e8f0' : '#1f2937'}">Painel Administrativo</h1>
                        <button onclick="window.openVehicleModal()" class="bg-green-500 text-white px-6 py-2 rounded-lg hover:bg-green-600 transition">
                            + Novo Veículo
                        </button>
                    </div>
                    <div class="bg-white rounded-xl shadow-lg overflow-hidden">
                        <div class="overflow-x-auto">
                            <table class="w-full">
                                <thead style="background:${isDarkMode ? '#334155' : '#f3f4f6'}">
                                    <tr>
                                        <th class="px-4 py-3 text-left text-sm font-medium" style="color:${isDarkMode ? '#cbd5e1' : '#374151'}">ID</th>
                                        <th class="px-4 py-3 text-left text-sm font-medium" style="color:${isDarkMode ? '#cbd5e1' : '#374151'}">Imagem</th>
                                        <th class="px-4 py-3 text-left text-sm font-medium" style="color:${isDarkMode ? '#cbd5e1' : '#374151'}">Veículo</th>
                                        <th class="px-4 py-3 text-left text-sm font-medium" style="color:${isDarkMode ? '#cbd5e1' : '#374151'}">Preço</th>
                                        <th class="px-4 py-3 text-left text-sm font-medium" style="color:${isDarkMode ? '#cbd5e1' : '#374151'}">Status</th>
                                        <th class="px-4 py-3 text-left text-sm font-medium" style="color:${isDarkMode ? '#cbd5e1' : '#374151'}">Fotos</th>
                                        <th class="px-4 py-3 text-left text-sm font-medium" style="color:${isDarkMode ? '#cbd5e1' : '#374151'}">Ações</th>
                                    </tr>
                                </thead>
                                <tbody>
                                    ${vehicles.map(v => `
                                        <tr class="border-t" style="border-color:${isDarkMode ? '#334155' : '#e5e7eb'}">
                                            <td class="px-4 py-3 text-sm">${v.id}</td>
                                            <td class="px-4 py-3"><img src="${v.imagens[0]}" class="w-12 h-12 object-cover rounded"></td>
                                            <td class="px-4 py-3"><div class="font-bold">${v.nome}</div><div class="text-sm" style="color:${isDarkMode ? '#94a3b8' : '#6b7280'}">${v.marca} ${v.ano}</div></td>
                                            <td class="px-4 py-3 text-sm font-semibold">${formatPrice(v.preco)}</td>
                                            <td class="px-4 py-3">
                                                <select onchange="window.updateStatus(${v.id}, this.value)" class="border rounded px-2 py-1 text-sm" style="background:${isDarkMode ? '#0f172a' : 'white'}; color:${isDarkMode ? '#e2e8f0' : '#1f2937'}; border-color:${isDarkMode ? '#334155' : '#d1d5db'}">
                                                    <option ${v.status === 'disponível' ? 'selected' : ''}>disponível</option>
                                                    <option ${v.status === 'reservado' ? 'selected' : ''}>reservado</option>
                                                    <option ${v.status === 'vendido' ? 'selected' : ''}>vendido</option>
                                                </select>
                                            </td>
                                            <td class="px-4 py-3 text-sm">📸 ${v.imagens.length}</td>
                                            <td class="px-4 py-3">
                                                <button onclick="window.editVehicle(${v.id})" class="text-blue-600 hover:text-blue-800 mr-3">Editar</button>
                                                <button onclick="handleDeleteVehicle(${v.id})" class="text-red-600 hover:text-red-800">Excluir</button>
                                            </td>
                                        </tr>
                                    `).join('')}
                                </tbody>
                            </table>
                        </div>
                    </div>
                </div>
            `;
        }
        
        html += `</div><footer class="bg-gray-800 text-white py-6 text-center"><p>© 2024 JM Veículos - Potência, estilo e confiança na sua garagem</p></footer></div>`;
        
        // Modal de Login
        if (showLoginModal) {
            html += `
                <div class="modal-overlay" onclick="window.closeLogin()">
                    <div class="modal-content" onclick="event.stopPropagation()">
                        <h2 class="text-2xl font-bold mb-4">Área Administrativa</h2>
                        <form onsubmit="window.submitLogin(event)">
                            <input type="email" id="loginEmail" placeholder="E-mail" class="w-full px-3 py-2 border rounded mb-3" required>
                            <input type="password" id="loginPass" placeholder="Senha" class="w-full px-3 py-2 border rounded mb-3" required>
                            <div class="p-3 rounded text-sm mb-4" style="background:${isDarkMode ? '#1e3a8a' : '#eff6ff'}">
                                <p class="font-bold">🔐 Credenciais:</p>
                                <p>E-mail: admin@jmveiculos.com</p>
                                <p>Senha: admin123</p>
                            </div>
                            <button type="submit" class="w-full bg-blue-500 text-white py-2 rounded hover:bg-blue-600">Entrar</button>
                            <button type="button" onclick="window.closeLogin()" class="w-full mt-2 bg-gray-500 text-white py-2 rounded hover:bg-gray-600">Cancelar</button>
                        </form>
                    </div>
                </div>
            `;
        }
        
        // Modal de Veículo
        if (showVehicleModal) {
            const v = editingVehicle || { nome: '', marca: MARCAS[0], ano: 2024, preco: 0, km: 0, combustivel: 'Flex', cambio: 'Manual', cor: '', descricao: '', imagens: [] };
            html += `
                <div class="modal-overlay" onclick="window.closeVehicleModal()">
                    <div class="modal-content" onclick="event.stopPropagation()">
                        <h2 class="text-2xl font-bold mb-4">${editingVehicle ? 'Editar Veículo' : 'Novo Veículo'}</h2>
                        <form onsubmit="window.submitVehicle(event)">
                            <input type="text" id="nome" placeholder="Nome do veículo" value="${v.nome}" class="w-full px-3 py-2 border rounded mb-3" required>
                            
                            <select id="marca" class="w-full px-3 py-2 border rounded mb-3">
                                ${MARCAS.map(m => `<option ${v.marca === m ? 'selected' : ''}>${m}</option>`).join('')}
                            </select>
                            
                            <div class="grid grid-cols-2 gap-3 mb-3">
                                <input type="number" id="ano" placeholder="Ano" value="${v.ano}" class="px-3 py-2 border rounded" required>
                                <input type="number" id="preco" placeholder="Preço" value="${v.preco}" class="px-3 py-2 border rounded" required>
                                <input type="number" id="km" placeholder="Quilometragem" value="${v.km}" class="px-3 py-2 border rounded" required>
                                <input type="text" id="cor" placeholder="Cor" value="${v.cor}" class="px-3 py-2 border rounded" required>
                            </div>
                            
                            <select id="combustivel" class="w-full px-3 py-2 border rounded mb-3">
                                <option ${v.combustivel === 'Flex' ? 'selected' : ''}>Flex</option>
                                <option ${v.combustivel === 'Gasolina' ? 'selected' : ''}>Gasolina</option>
                                <option ${v.combustivel === 'Diesel' ? 'selected' : ''}>Diesel</option>
                                <option ${v.combustivel === 'Etanol' ? 'selected' : ''}>Etanol</option>
                            </select>
                            
                            <select id="cambio" class="w-full px-3 py-2 border rounded mb-3">
                                <option ${v.cambio === 'Manual' ? 'selected' : ''}>Manual</option>
                                <option ${v.cambio === 'Automático' ? 'selected' : ''}>Automático</option>
                                <option ${v.cambio === 'CVT' ? 'selected' : ''}>CVT</option>
                            </select>
                            
                            <textarea id="descricao" placeholder="Descrição completa" rows="3" class="w-full px-3 py-2 border rounded mb-3" required>${v.descricao}</textarea>
                            
                            <div class="mb-3">
                                <label class="block text-sm font-medium mb-2">Fotos do Veículo (adicione quantas quiser)</label>
                                <div id="imagePreview" class="gallery-preview">
                                    ${v.imagens.map(img => `
                                        <div class="gallery-item">
                                            <img src="${img}" class="preview-image">
                                            <div class="remove-image-btn" onclick="this.parentElement.remove()">×</div>
                                        </div>
                                    `).join('')}
                                </div>
                                <input type="file" id="vehicleImages" accept="image/*" multiple style="display:none">
                                <button type="button" onclick="document.getElementById('vehicleImages').click()" class="add-image-btn">+ Adicionar Fotos</button>
                            </div>
                            
                            <div class="flex gap-3 mt-4">
                                <button type="submit" class="flex-1 bg-blue-500 text-white py-2 rounded hover:bg-blue-600">Salvar</button>
                                <button type="button" onclick="window.closeVehicleModal()" class="flex-1 bg-gray-500 text-white py-2 rounded hover:bg-gray-600">Cancelar</button>
                            </div>
                        </form>
                    </div>
                </div>
            `;
        }
        
        if (toastMessage) {
            html += `<div class="toast" style="background:${toastMessage.isError ? '#ef4444' : '#10b981'}">${toastMessage.text}</div>`;
        }
        
        app.innerHTML = html;
        
        // Configurar evento de upload de fotos após renderizar
        if (showVehicleModal) {
            const uploadInput = document.getElementById('vehicleImages');
            if (uploadInput) {
                uploadInput.onchange = function(e) {
                    if (e.target.files && e.target.files.length > 0) {
                        Array.from(e.target.files).forEach((file) => {
                            const reader = new FileReader();
                            reader.onload = (event) => {
                                const previewDiv = document.getElementById('imagePreview');
                                const imgDiv = document.createElement('div');
                                imgDiv.className = 'gallery-item';
                                imgDiv.innerHTML = `
                                    <img src="${event.target.result}" class="preview-image">
                                    <div class="remove-image-btn" onclick="this.parentElement.remove()">×</div>
                                `;
                                previewDiv.appendChild(imgDiv);
                            };
                            reader.readAsDataURL(file);
                        });
                    }
                };
            }
        }
    }
    
    // Funções globais
    window.changePage = (page) => { currentPage = page; selectedVehicle = null; render(); };
    window.viewDetail = (id) => { 
        selectedVehicle = vehicles.find(v => v.id === id); 
        currentImageIndex = 0; 
        currentPage = 'detail'; 
        render(); 
    };
    window.setImageIndex = (i) => { currentImageIndex = i; render(); };
    window.updateStatus = (id, status) => { 
        const v = vehicles.find(v => v.id === id); 
        if (v) { 
            v.status = status; 
            saveData(); 
            render(); 
            showToast('Status atualizado!'); 
        } 
    };
    window.showLogin = () => { showLoginModal = true; render(); };
    window.closeLogin = () => { showLoginModal = false; render(); };
    window.submitLogin = (e) => { e.preventDefault(); handleLogin(document.getElementById('loginEmail').value, document.getElementById('loginPass').value); };
    window.openVehicleModal = () => { editingVehicle = null; showVehicleModal = true; render(); };
    window.closeVehicleModal = () => { showVehicleModal = false; editingVehicle = null; render(); };
    window.editVehicle = (id) => { editingVehicle = JSON.parse(JSON.stringify(vehicles.find(v => v.id === id))); showVehicleModal = true; render(); };
    
    window.submitVehicle = async (e) => {
        e.preventDefault();
        
        const nome = document.getElementById('nome').value;
        const marca = document.getElementById('marca').value;
        const ano = parseInt(document.getElementById('ano').value);
        const preco = parseInt(document.getElementById('preco').value);
        const km = parseInt(document.getElementById('km').value);
        const cor = document.getElementById('cor').value;
        const combustivel = document.getElementById('combustivel').value;
        const cambio = document.getElementById('cambio').value;
        const descricao = document.getElementById('descricao').value;
        
        // Coletar imagens da pré-visualização
        const previewDiv = document.getElementById('imagePreview');
        const imagens = [];
        if (previewDiv) {
            const imgElements = previewDiv.querySelectorAll('.gallery-item img');
            imgElements.forEach(img => {
                if (img.src) imagens.push(img.src);
            });
        }
        
        if (editingVehicle) {
            const index = vehicles.findIndex(v => v.id === editingVehicle.id);
            if (index !== -1) {
                vehicles[index] = { 
                    ...editingVehicle, 
                    nome, marca, ano, preco, km, cor, combustivel, cambio, descricao, 
                    imagens: imagens.length ? imagens : editingVehicle.imagens 
                };
                saveData();
                showToast('Veículo atualizado com sucesso!');
            }
        } else {
            const newVehicle = { 
                id: Date.now(), 
                nome, marca, ano, preco, km, cor, combustivel, cambio, descricao, 
                status: 'disponível', 
                imagens: imagens.length ? imagens : ['https://placehold.co/800x450?text=Sem+Imagem'] 
            };
            vehicles.push(newVehicle);
            saveData();
            showToast('Veículo adicionado com sucesso!');
        }
        showVehicleModal = false;
        editingVehicle = null;
        render();
    };
    
    // Inicializar
    loadData();
    render();
</script>
